    </style>

<body class="bg-gray-100 min-h-screen p-4 pb-64">

    <div id="app" class="max-w-md mx-auto">

        <h1 class="text-2xl font-bold mb-4 text-center">飯店布品登記</h1>

        <div id="loading" class="text-center py-10 text-gray-500">正在嘗試讀取資料...</div>

       

        <div id="content" class="hidden">

            <div class="bg-white p-4 rounded-xl shadow-sm mb-4 space-y-2">

                <input type="date" id="datePicker" class="w-full p-3 bg-gray-100 rounded-lg text-xl font-bold">

                <input type="text" id="search" placeholder="🔍 搜尋關鍵字..." class="w-full p-3 border rounded-lg text-xl font-bold" oninput="filterList()">

                <select id="hotelSelect" class="w-full p-3 bg-gray-100 rounded-lg text-xl font-bold" onchange="filterList()"></select>

            </div>

            <div id="itemsList"></div>

        </div>

    </div>

    <script>

        const API_URL = "https://script.google.com/macros/s/AKfycbwJv-qdQcoqowK1JYqSttVGYiSh0nW-SO1-6N_D9pSHok36i7XMt_uL_oRMKJGxFWw/exec";

       

        async function init() {

            try {

                document.getElementById('datePicker').valueAsDate = new Date();

                const res = await fetch(API_URL + "?action=getData");

                const allItems = await res.json();

               

                if(allItems.error) throw new Error(allItems.error);

                const validItems = allItems.filter(d => d.hotel && d.item);

                const hotels = [...new Set(validItems.map(d => d.hotel))];

                document.getElementById('hotelSelect').innerHTML = hotels.map(h => `<option value="${h}">${h}</option>`).join('');

               

                validItems.forEach(d => createItemRow(d.hotel, d.item));

               

                document.getElementById('loading').classList.add('hidden');

                document.getElementById('content').classList.remove('hidden');

                filterList();

            } catch (err) {

                document.getElementById('loading').innerHTML = `<p class="text-red-500 font-bold">錯誤: ${err.message}</p>`;

            }

        }

        function createItemRow(hotel, item) {

            const list = document.getElementById('itemsList');

            const div = document.createElement('div');

            div.className = 'item-row';

            div.dataset.hotel = hotel;

            div.dataset.item = item.toLowerCase();

            div.innerHTML = `

                <div class="flex justify-between items-center mb-1">

                    <span class="text-2xl font-black text-gray-900">${item}</span>

                    <input type="text" class="w-32 p-2 text-center text-3xl font-bold border-b-2 border-blue-500" placeholder="0">

                </div>

                <div class="flex gap-2 items-center">

                    <input type="text" class="flex-1 p-1.5 bg-gray-50 rounded text-sm note-input" placeholder="備註...">

                </div>

            `;

            list.appendChild(div);

        }

        function filterList() {

            const hotel = document.getElementById('hotelSelect').value;

            const search = document.getElementById('search').value.toLowerCase();

            document.querySelectorAll('.item-row').forEach(row => {

                row.style.display = (row.dataset.hotel === hotel && row.dataset.item.includes(search)) ? 'block' : 'none';

            });

        }

       

        init();

    </script>
